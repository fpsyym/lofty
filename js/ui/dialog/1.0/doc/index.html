对话框组件

使用方法：单独使用buttons回调【实例：普通弹出框DPL】

define(['fui/dialog/1.0'], function(Dialog){
	var dialog = new Dialog({
			isModal:true,
			dragAble:true,
			dragSelector:".d-header",
			buttons:[
				{
					'event':'click',//事件类型
					'selector':'.fui-btn1',//要绑定事件的元素
					'callback':function(paramObj){//事件的回调函数
						alert(paramObj.name);
						console.log(this);//这里的this指向的是当前的dialog
					},
					'paramObj':{//回调函数要传入的参数(封装成对象便于管理)
						'name':'按钮1'
					}
				},
				{
					'event':'click',
					'selector':'.fui-btn2',
					'callback':function(paramObj){
						alert(paramObj.name);
					},
					'paramObj':{
						'name':'按钮2'
					}
				},
				{
					'event':'click',
					'selector':'.d-close',
					'callback':function(paramObj){
						this.hide();
					}
				}
			]
	});
	//dialog显示
	dialog.show({
			selector:'.fui-dialog-body'
	});
});

使用方法：单独使用timers回调【实例：弹出消息浮层DPL】

define(['fui/dialog/1.0'], function(Dialog){
	var dialog = new Dialog({
			isModal:true,
			dragAble:true,
			dragSelector:".d-header",
			timers:[
				{
					'time':2000,//延迟时间单位毫秒
					'callback':function(paramObj){
						this.hide();//this指向的是当前dialog组件
					},
					'paramObj':{//回调函数要传入的参数(封装成对象便于管理)
						'name':'timer'
					}
				}
			]
	});
	//dialog显示
	dialog.show({
			selector:'.fui-dialog-body'
	});
});

配置参数

配置参数 参数类型/可选值 参数默认值 作用说明
x number/可选 无 弹出框X轴的位置
y number/可选 无 弹出框Y轴的位置
zindex Number 1688 弹出层的z-index
isModal boolean false 是否锁屏
dragAble boolean false 是否可以被拖拽
dragSelector string/可选 false dragAble为true时，指定按下哪一个元素开始拖拽
buttons [{object},{object},...] 无 给组件内部的元素绑定事件和回调【详细情况，见使用方法】
timers [{object},{object},...] 无 给组件设置延时回调的函数【详细情况，见使用方法】

公共方法

方法名称 入参说明 返回值 作用说明
show ({object}) {config} / 设置dialog显示，有两种调用方式:
1、show({txt:'html代码片段'});
2、show({selector:'.selector'});
hide () / / 隐藏dialog
