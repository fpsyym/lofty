使用方法

define(['fui/flash/1.0'], function(Flash){
	//所有的参数配置都将反映到flash的param上
	var flash = new Flash({
		container:"#test",
		swf: 'test_flashvars.swf',
		flashvars: {
			name1: 'jQuery',
			name2: 'jQuery.ui.flash',
			name3: 'Denis'
		}
	});
});

Flash.util.version.release / number 数字格式的版本信息
